using System.Net;
using System.Net.Mail;

var client = new SmtpClient("smtp.yourserver.com", 587)
{
    Credentials = new NetworkCredential("your_email", "your_password"),
    EnableSsl = true
};

var mail = new MailMessage("your_email", "recipient@example.com", "Уведомление", "Ваша лицензия скоро истекает");
client.Send(mail);
